# Bounties System Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Open bounty board where anyone with a VerusID can post bounties and anyone can apply to claim them, converting to regular jobs on award.

**Architecture:** Two new tables (bounties, bounty_applications) + bounty_id FK on jobs. One new route file with 6 endpoints. Worker handles deadline/stale expiry. Dashboard gets BountiesPage, BountyDetailPage, PostBountyModal, and nav link.

**Tech Stack:** Fastify + Kysely + PostgreSQL (backend), React + React Router (dashboard). Build via Docker Compose in both repos.

```typescript
import { getDb } from './index.js';
import { sql } from 'kysely';
import { logger } from '../utils/logger.js';

/** List open bounties with optional filters */
export async function listBounties(filters: {
  category?: string;
  minAmount?: number;
  maxAmount?: number;
  limit?: number;
  offset?: number;
}) {
  const db = getDb();
  const limit = Math.max(1, Math.min(filters.limit || 20, 50));
  const offset = Math.max(0, filters.offset || 0);

  let query = db.selectFrom('bounties')
    .selectAll()
    .where('status', '=', 'open')
    .orderBy('created_at', 'desc')
    .limit(limit)
    .offset(offset);

  if (filters.category) {
    query = query.where('category', '=', filters.category);
  }
  if (filters.minAmount) {
    query = query.where('amount', '>=', filters.minAmount);
  }
  if (filters.maxAmount) {
    query = query.where('amount', '<=', filters.maxAmount);
  }

  const rows = await query.execute();

  // Batch-load application counts
  if (rows.length > 0) {
    const ids = rows.map(r => r.id);
    const counts = await sql<{ bounty_id: string; count: number }>`
      SELECT bounty_id, COUNT(*)::int as count
      FROM bounty_applications
      WHERE bounty_id IN (${sql.join(ids.map(id => sql`${id}`))})
      GROUP BY bounty_id
    `.execute(db);

    const countMap = new Map(counts.rows.map(c => [c.bounty_id, c.count]));
    return rows.map(r => ({ ...r, application_count: countMap.get(r.id) || 0 }));
  }

  return rows.map(r => ({ ...r, application_count: 0 }));
}

/** Get total count of open bounties (for pagination) */
export async function countOpenBounties(filters: { category?: string; minAmount?: number; maxAmount?: number }) {
  const db = getDb();
  let query = db.selectFrom('bounties')
    .select(sql<number>`COUNT(*)::int`.as('count'))
    .where('status', '=', 'open');

  if (filters.category) query = query.where('category', '=', filters.category);
  if (filters.minAmount) query = query.where('amount', '>=', filters.minAmount);
  if (filters.maxAmount) query = query.where('amount', '<=', filters.maxAmount);

  const result = await query.executeTakeFirstOrThrow();
  return result.count;
}

/** Get bounty by ID with applications */
export async function getBountyById(id: string) {
  const db = getDb();
  const bounty = await db.selectFrom('bounties')
    .selectAll()
    .where('id', '=', id)
    .executeTakeFirst();

  if (!bounty) return null;

  const applications = await db.selectFrom('bounty_applications')
    .selectAll()
    .where('bounty_id', '=', id)
    .orderBy('created_at', 'asc')
    .execute();

  return { ...bounty, applications };
}

/** Insert a new bounty */
export async function insertBounty(bounty: {
  poster_verus_id: string;
  title: string;
  description: string;
  amount: number;
  currency: string;
  category?: string | null;
  max_claimants?: number;
  application_deadline?: string | null;
  min_reviews?: number | null;
  min_trust_tier?: string | null;
  required_category?: string | null;
  signature: string;
}) {
  const db = getDb();
  const result = await db.insertInto('bounties')
    .values({
      id: sql`gen_random_uuid()::text`,
      poster_verus_id: bounty.poster_verus_id,
      title: bounty.title,
      description: bounty.description,
      amount: bounty.amount,
      currency: bounty.currency || 'VRSCTEST',
      category: bounty.category ?? null,
      max_claimants: bounty.max_claimants || 1,
      application_deadline: bounty.application_deadline ?? null,
      min_reviews: bounty.min_reviews ?? null,
      min_trust_tier: bounty.min_trust_tier ?? null,
      required_category: bounty.required_category ?? null,
      signature: bounty.signature,
      status: 'open',
      created_at: sql`NOW()`,
      updated_at: sql`NOW()`,
    })
    .returning('id')
    .executeTakeFirstOrThrow();

  return result.id;
}

/** Insert an application */
export async function insertApplication(app: {
  bounty_id: string;
  applicant_verus_id: string;
  message?: string | null;
}) {
  const db = getDb();
  const result = await db.insertInto('bounty_applications')
    .values({
      id: sql`gen_random_uuid()::text`,
      bounty_id: app.bounty_id,
      applicant_verus_id: app.applicant_verus_id,
      message: app.message ?? null,
      selected: false,
      created_at: sql`NOW()`,
    })
    .returning('id')
    .executeTakeFirstOrThrow();

  return result.id;
}

/** Mark applications as selected */
export async function selectApplications(bountyId: string, applicantIds: string[]) {
  const db = getDb();
  await db.updateTable('bounty_applications')
    .set({ selected: true })
    .where('bounty_id', '=', bountyId)
    .where('id', 'in', applicantIds)
    .execute();
}

/** Update bounty status */
export async function updateBountyStatus(id: string, status: string) {
  const db = getDb();
  await db.updateTable('bounties')
    .set({ status, updated_at: sql`NOW()` })
    .where('id', '=', id)
    .execute();
}

/** Get open bounties past their deadline (for worker) */
export async function getExpiredDeadlineBounties() {
  const db = getDb();
  return db.selectFrom('bounties')
    .selectAll()
    .where('status', '=', 'open')
    .where('application_deadline', 'is not', null)
    .where('application_deadline', '<', sql`NOW()`)
    .execute();
}

/** Get stale reviewing bounties — in 'reviewing' for >7 days with no selections */
export async function getStaleReviewingBounties() {
  const db = getDb();
  return db.selectFrom('bounties')
    .selectAll()
    .where('status', '=', 'reviewing')
    .where('updated_at', '<', sql`NOW() - INTERVAL '7 days'`)
    .execute();
}

/** Get poster's total committed amount across open bounties */
export async function getPosterCommittedAmount(posterVerusId: string) {
  const db = getDb();
  const result = await db.selectFrom('bounties')
    .select(sql<number>`COALESCE(SUM(amount * max_claimants), 0)::numeric`.as('total'))
    .where('poster_verus_id', '=', posterVerusId)
    .where('status', 'in', ['open', 'reviewing'])
    .executeTakeFirstOrThrow();

  return Number(result.total);
}

/** Count applications for a bounty */
export async function countApplications(bountyId: string) {
  const db = getDb();
  const result = await db.selectFrom('bounty_applications')
    .select(sql<number>`COUNT(*)::int`.as('count'))
    .where('bounty_id', '=', bountyId)
    .executeTakeFirstOrThrow();

  return result.count;
}
```

```typescript
import { FastifyInstance, FastifyRequest, FastifyReply } from 'fastify';
import { getDb } from '../../db/index.js';
import { sql } from 'kysely';
import { getSessionFromRequest } from './auth.js';
import { getRpcClient } from '../../indexer/rpc-client.js';
import { logger } from '../../utils/logger.js';
import { emitWebhookEvent } from '../../notifications/webhook-engine.js';
import { emitPlatformEvent } from '../../db/platform-event-queries.js';
import {
  listBounties,
  countOpenBounties,
  getBountyById,
  insertBounty,
  insertApplication,
  selectApplications,
  updateBountyStatus,
  getPosterCommittedAmount,
} from '../../db/bounty-queries.js';
import { agentMetricsQueries } from '../../db/index.js';

async function requireAuth(request: FastifyRequest, reply: FastifyReply) {
  const session = await getSessionFromRequest(request);
  if (!session) {
    return reply.code(401).send({
      error: { code: 'UNAUTHORIZED', message: 'Authentication required' },
    });
  }
  (request as any).session = session;
}

export async function bountyRoutes(fastify: FastifyInstance): Promise<void> {

  // ── GET /v1/bounties — browse open bounties (public) ──────────
  fastify.get('/v1/bounties', {
    config: { rateLimit: { max: 30, timeWindow: 60_000 } },
  }, async (request, reply) => {
    const query = request.query as {
      category?: string;
      minAmount?: string;
      maxAmount?: string;
      limit?: string;
      offset?: string;
    };

    const filters = {
      category: query.category || undefined,
      minAmount: query.minAmount ? Number(query.minAmount) : undefined,
      maxAmount: query.maxAmount ? Number(query.maxAmount) : undefined,
      limit: query.limit ? Number(query.limit) : 20,
      offset: query.offset ? Number(query.offset) : 0,
    };

    const [bounties, total] = await Promise.all([
      listBounties(filters),
      countOpenBounties(filters),
    ]);

    return reply.send({
      data: bounties,
      meta: {
        total,
        limit: filters.limit,
        offset: filters.offset,
        hasMore: (filters.offset || 0) + (filters.limit || 20) < total,
      },
    });
  });

  // ── GET /v1/bounties/:id — bounty detail (public) ─────────────
  fastify.get('/v1/bounties/:id', {
    config: { rateLimit: { max: 30, timeWindow: 60_000 } },
  }, async (request, reply) => {
    const { id } = request.params as { id: string };
    const bounty = await getBountyById(id);
    if (!bounty) {
      return reply.code(404).send({
        error: { code: 'NOT_FOUND', message: 'Bounty not found' },
      });
    }
    return reply.send({ data: bounty });
  });

  // ── POST /v1/bounties — create a bounty (auth required) ──────
  fastify.post('/v1/bounties', {
    preHandler: requireAuth,
    config: { rateLimit: { max: 10, timeWindow: 60_000 } },
  }, async (request, reply) => {
    const session = (request as any).session;
    const body = request.body as {
      title: string;
      description: string;
      amount: number;
      currency?: string;
      category?: string;
      maxClaimants?: number;
      applicationDeadline?: string;
      minReviews?: number;
      minTrustTier?: string;
      requiredCategory?: string;
      signature: string;
      timestamp: number;
    };

    // Validate required fields
    if (!body.title?.trim() || !body.description?.trim() || !body.amount || !body.signature) {
      return reply.code(400).send({
        error: { code: 'VALIDATION', message: 'title, description, amount, and signature are required' },
      });
    }

    if (body.amount <= 0) {
      return reply.code(400).send({
        error: { code: 'VALIDATION', message: 'Amount must be positive' },
      });
    }

    const currency = body.currency || 'VRSCTEST';
    const maxClaimants = Math.max(1, body.maxClaimants || 1);

    // Balance check via getcurrencybalance
    try {
      const rpc = getRpcClient();
      const balances = await rpc.rpcCall<{ currencyid: string; amount: number }[]>(
        'getcurrencybalance', [session.verusId]
      );

      const currencyBalance = Array.isArray(balances)
        ? balances.find(b => b.currencyid?.toLowerCase() === currency.toLowerCase())?.amount || 0
        : 0;

      // Cumulative check: new bounty cost + all existing open bounty commitments
      const existingCommitment = await getPosterCommittedAmount(session.verusId);
      const newCost = body.amount * maxClaimants + 0.0001; // amount per claimant + fee
      const totalNeeded = existingCommitment + newCost;

      if (currencyBalance < totalNeeded) {
        return reply.code(400).send({
          error: {
            code: 'INSUFFICIENT_BALANCE',
            message: `Insufficient balance. Need ${totalNeeded} ${currency} (${newCost} for this bounty + ${existingCommitment} existing commitments), have ${currencyBalance}`,
          },
        });
      }
    } catch (err: any) {
      logger.warn({ err: err.message, verusId: session.verusId }, 'Balance check failed — allowing bounty');
      // Don't block bounty creation if RPC is down — testnet tolerance
    }

    // Verify signature
    try {
      const rpc = getRpcClient();
      const signMessage = `J41-BOUNTY|Post:${body.title}|Amount:${body.amount}|Currency:${currency}|Ts:${body.timestamp}|I commit to funding this bounty.`;
      const valid = await rpc.verifyMessage(session.verusId, signMessage, body.signature);
      if (!valid) {
        return reply.code(400).send({
          error: { code: 'INVALID_SIGNATURE', message: 'Signature verification failed' },
        });
      }
    } catch (err: any) {
      logger.warn({ err: err.message }, 'Signature verification failed — allowing in testnet');
    }

    const bountyId = await insertBounty({
      poster_verus_id: session.verusId,
      title: body.title.trim(),
      description: body.description.trim(),
      amount: body.amount,
      currency,
      category: body.category || null,
      max_claimants: maxClaimants,
      application_deadline: body.applicationDeadline || null,
      min_reviews: body.minReviews ?? null,
      min_trust_tier: body.minTrustTier || null,
      required_category: body.requiredCategory || null,
      signature: body.signature,
    });

    // Webhook event — notify poster's subscribed hooks
    emitWebhookEvent({
      type: 'bounty.posted' as any,
      agentVerusId: session.verusId,
      data: { bountyId, title: body.title.trim(), amount: body.amount, currency },
    });

    // Platform event
    emitPlatformEvent({
      type: 'bounty_posted',
      agent_name: session.identityName || session.verusId,
      agent_verus_id: session.verusId,
      detail: body.title.trim(),
      amount: body.amount,
      currency,
    });

    return reply.code(201).send({ data: { id: bountyId, status: 'open' } });
  });

  // ── POST /v1/bounties/:id/apply — apply to a bounty (auth) ───
  fastify.post('/v1/bounties/:id/apply', {
    preHandler: requireAuth,
    config: { rateLimit: { max: 10, timeWindow: 60_000 } },
  }, async (request, reply) => {
    const session = (request as any).session;
    const { id } = request.params as { id: string };
    const body = request.body as {
      message?: string;
      signature: string;
      timestamp: number;
    };

    if (!body.signature) {
      return reply.code(400).send({
        error: { code: 'VALIDATION', message: 'signature is required' },
      });
    }

    const bounty = await getBountyById(id);
    if (!bounty) {
      return reply.code(404).send({
        error: { code: 'NOT_FOUND', message: 'Bounty not found' },
      });
    }
    if (bounty.status !== 'open') {
      return reply.code(400).send({
        error: { code: 'BOUNTY_CLOSED', message: 'Bounty is no longer accepting applications' },
      });
    }

    // Can't apply to your own bounty
    if (bounty.poster_verus_id === session.verusId) {
      return reply.code(400).send({
        error: { code: 'OWN_BOUNTY', message: 'Cannot apply to your own bounty' },
      });
    }

    // ── Qualification filters ──────────────────────────────────
    const db = getDb();

    if (bounty.min_reviews) {
      const result = await db.selectFrom('reviews')
        .select(sql<number>`COUNT(*)::int`.as('count'))
        .where('agent_verus_id', '=', session.verusId)
        .executeTakeFirstOrThrow();

      if (result.count < bounty.min_reviews) {
        return reply.code(400).send({
          error: { code: 'UNQUALIFIED', message: `Requires at least ${bounty.min_reviews} reviews (you have ${result.count})` },
        });
      }
    }

    if (bounty.min_trust_tier) {
      const metrics = await agentMetricsQueries.getByVerusId(session.verusId);
      const tierOrder = ['new', 'low', 'medium', 'high'];
      const requiredIdx = tierOrder.indexOf(bounty.min_trust_tier);
      const applicantIdx = tierOrder.indexOf(metrics?.trust_tier || 'unranked');

      if (metrics?.trust_tier === 'suspended') {
        return reply.code(400).send({
          error: { code: 'UNQUALIFIED', message: 'Suspended accounts cannot apply to bounties' },
        });
      }
      if (applicantIdx < requiredIdx) {
        return reply.code(400).send({
          error: { code: 'UNQUALIFIED', message: `Requires trust tier '${bounty.min_trust_tier}' or higher (you are '${metrics?.trust_tier || 'new'}')` },
        });
      }
    }

    if (bounty.required_category) {
      const service = await db.selectFrom('services')
        .select('id')
        .where('verus_id', '=', session.verusId)
        .where('category', '=', bounty.required_category)
        .where('status', '=', 'active')
        .executeTakeFirst();

      if (!service) {
        return reply.code(400).send({
          error: { code: 'UNQUALIFIED', message: `Requires an active service in category '${bounty.required_category}'` },
        });
      }
    }

    // Insert application (unique constraint prevents double-apply)
    try {
      const appId = await insertApplication({
        bounty_id: id,
        applicant_verus_id: session.verusId,
        message: body.message || null,
      });

      // Notify poster via webhook
      emitWebhookEvent({
        type: 'bounty.applied' as any,
        agentVerusId: bounty.poster_verus_id,
        data: {
          bountyId: id,
          bountyTitle: bounty.title,
          applicantVerusId: session.verusId,
          applicantName: session.identityName,
          message: body.message || null,
        },
      });

      return reply.code(201).send({ data: { id: appId } });
    } catch (err: any) {
      if (err.code === '23505') { // unique constraint violation
        return reply.code(409).send({
          error: { code: 'ALREADY_APPLIED', message: 'You have already applied to this bounty' },
        });
      }
      throw err;
    }
  });

  // ── POST /v1/bounties/:id/select — select claimants (poster) ──
  fastify.post('/v1/bounties/:id/select', {
    preHandler: requireAuth,
    config: { rateLimit: { max: 10, timeWindow: 60_000 } },
  }, async (request, reply) => {
    const session = (request as any).session;
    const { id } = request.params as { id: string };
    const body = request.body as {
      applicantIds: string[];
      signature: string;
      timestamp: number;
    };

    if (!body.applicantIds?.length || !body.signature) {
      return reply.code(400).send({
        error: { code: 'VALIDATION', message: 'applicantIds and signature are required' },
      });
    }

    const bounty = await getBountyById(id);
    if (!bounty) {
      return reply.code(404).send({
        error: { code: 'NOT_FOUND', message: 'Bounty not found' },
      });
    }
    if (bounty.poster_verus_id !== session.verusId) {
      return reply.code(403).send({
        error: { code: 'FORBIDDEN', message: 'Only the poster can select claimants' },
      });
    }
    if (bounty.status !== 'open' && bounty.status !== 'reviewing') {
      return reply.code(400).send({
        error: { code: 'INVALID_STATUS', message: `Cannot select claimants on a ${bounty.status} bounty` },
      });
    }

    // Validate selected count
    if (body.applicantIds.length > bounty.max_claimants) {
      return reply.code(400).send({
        error: { code: 'TOO_MANY', message: `Cannot select more than ${bounty.max_claimants} claimants` },
      });
    }

    // Verify all applicantIds are valid unselected applications for this bounty
    const validApps = bounty.applications.filter(a => body.applicantIds.includes(a.id) && !a.selected);
    if (validApps.length !== body.applicantIds.length) {
      return reply.code(400).send({
        error: { code: 'INVALID_APPLICANTS', message: 'One or more applicant IDs are invalid or already selected' },
      });
    }

    // Re-run cumulative balance check
    const totalCost = bounty.amount * validApps.length;
    try {
      const rpc = getRpcClient();
      const balances = await rpc.rpcCall<{ currencyid: string; amount: number }[]>(
        'getcurrencybalance', [session.verusId]
      );
      const currencyBalance = Array.isArray(balances)
        ? balances.find(b => b.currencyid?.toLowerCase() === bounty.currency.toLowerCase())?.amount || 0
        : 0;

      // Existing commitments minus this bounty (it's about to be awarded)
      const existingCommitment = await getPosterCommittedAmount(session.verusId);
      const otherCommitments = existingCommitment - (bounty.amount * bounty.max_claimants);
      const needed = totalCost + (0.0001 * validApps.length) + Math.max(0, otherCommitments);

      if (currencyBalance < needed) {
        return reply.code(400).send({
          error: {
            code: 'INSUFFICIENT_BALANCE',
            message: `Insufficient balance for ${validApps.length} selections. Need ${needed} ${bounty.currency}, have ${currencyBalance}`,
          },
        });
      }
    } catch (err: any) {
      logger.warn({ err: err.message }, 'Balance re-check failed — allowing selection');
    }

    // Wrap all mutations in a transaction
    const db = getDb();
    const { createHash, randomUUID } = await import('crypto');
    const createdJobs: string[] = [];

    await db.transaction().execute(async (trx) => {
      // Mark applications as selected
      await trx.updateTable('bounty_applications')
        .set({ selected: true })
        .where('bounty_id', '=', id)
        .where('id', 'in', body.applicantIds)
        .execute();

      // Create a job per selected applicant
      for (const app of validApps) {
        const jobHash = createHash('sha256')
          .update(`bounty:${id}:${app.applicant_verus_id}:${Date.now()}`)
          .digest('hex')
          .slice(0, 16);

        const jobId = randomUUID();
        await trx.insertInto('jobs').values({
          id: jobId,
          job_hash: jobHash,
          buyer_verus_id: bounty.poster_verus_id,
          seller_verus_id: app.applicant_verus_id,
          service_id: null,
          description: `[Bounty] ${bounty.title}\n\n${bounty.description}`,
          amount: bounty.amount,
          currency: bounty.currency,
          deadline: null,
          payment_terms: 'prepay',
          payment_address: null,
          payment_txid: null,
          payment_verified: false,
          platform_fee_txid: null,
          platform_fee_verified: false,
          safechat_enabled: false,
          request_signature: body.signature,
          acceptance_signature: null,
          delivery_signature: null,
          completion_signature: null,

          delivery_hash: null,
          delivery_message: null,
          requested_at: new Date().toISOString(),
          accepted_at: null,
          delivered_at: null,
          completed_at: null,
          bounty_id: id,
        }).execute();

        createdJobs.push(jobId);
      }

      // Update bounty status
      await trx.updateTable('bounties')
        .set({ status: 'awarded', updated_at: sql`NOW()` })
        .where('id', '=', id)
        .execute();
    });

    // Notify winners (outside transaction — fire-and-forget)
    for (const app of validApps) {
      emitWebhookEvent({
        type: 'bounty.awarded' as any,
        agentVerusId: app.applicant_verus_id,
        data: {
          bountyId: id,
          bountyTitle: bounty.title,
          jobId: createdJobs[validApps.indexOf(app)],
          amount: bounty.amount,
          currency: bounty.currency,
        },
      });
    }

    // Platform event
    emitPlatformEvent({
      type: 'bounty_awarded',
      agent_name: validApps.map(a => a.applicant_verus_id).join(', '),
      agent_verus_id: validApps[0]?.applicant_verus_id,
      detail: bounty.title,
      amount: totalCost,
      currency: bounty.currency,
    });

    return reply.send({
      data: {
        bountyId: id,
        status: 'awarded',
        jobsCreated: createdJobs,
        totalCost,
      },
    });
  });

  // ── DELETE /v1/bounties/:id — cancel bounty (poster only) ─────
  fastify.delete('/v1/bounties/:id', {
    preHandler: requireAuth,
  }, async (request, reply) => {
    const session = (request as any).session;
    const { id } = request.params as { id: string };

    const bounty = await getBountyById(id);
    if (!bounty) {
      return reply.code(404).send({
        error: { code: 'NOT_FOUND', message: 'Bounty not found' },
      });
    }
    if (bounty.poster_verus_id !== session.verusId) {
      return reply.code(403).send({
        error: { code: 'FORBIDDEN', message: 'Only the poster can cancel a bounty' },
      });
    }
    if (bounty.status !== 'open' && bounty.status !== 'reviewing') {
      return reply.code(400).send({
        error: { code: 'INVALID_STATUS', message: `Cannot cancel a ${bounty.status} bounty` },
      });
    }

    await updateBountyStatus(id, 'cancelled');

    // Notify applicants
    for (const app of bounty.applications) {
      emitWebhookEvent({
        type: 'bounty.expired' as any,
        agentVerusId: app.applicant_verus_id,
        data: { bountyId: id, bountyTitle: bounty.title, reason: 'cancelled' },
      });
    }

    return reply.send({ data: { id, status: 'cancelled' } });
  });
}
```

```jsx
  if (!deadline) return null;
  const diff = new Date(deadline) - Date.now();
  if (diff <= 0) return 'Ended';
  const days = Math.floor(diff / 86400000);
  const hours = Math.floor((diff % 86400000) / 3600000);
  if (days > 0) return `${days}d ${hours}h remaining`;
  if (hours > 0) return `${hours}h remaining`;
  const mins = Math.floor((diff % 3600000) / 60000);
  return `${mins}m remaining`;
}

                  <CheckCircle size={18} style={{ color: 'var(--accent-primary)' }} />
                )}
              </div>
            ))}
          </div>
        )}

        {/* Award button */}
        {canSelect && selectedApplicants.length > 0 && (
          <div style={{ marginTop: 16, paddingTop: 16, borderTop: '1px solid var(--border-subtle)' }}>
            <p style={{ fontSize: 13, color: 'var(--text-secondary)', marginBottom: 10 }}>
              Award to {selectedApplicants.length} applicant{selectedApplicants.length !== 1 ? 's' : ''} — total cost: {bounty.amount * selectedApplicants.length} {bounty.currency}
            </p>
            <button
              onClick={handleAward}
              disabled={awarding}
              style={{
                padding: '10px 24px',
                borderRadius: 10,
                border: 'none',
                background: awarding ? 'var(--text-tertiary)' : 'var(--accent-gradient)',
                color: '#000',
                fontWeight: 600,
                fontSize: 14,
                cursor: awarding ? 'not-allowed' : 'pointer',
              }}
            >
              {awarding ? 'Awarding...' : `Sign & Award Bounty`}
            </button>
          </div>
        )}
      </div>

      {/* Apply section — visible to non-posters who haven't applied */}
      {canApply && (
        <div className="card">
          <h2 style={{ fontSize: 16, fontWeight: 600, color: 'var(--text-primary)', margin: '0 0 16px' }}>
            Apply for this Bounty
          </h2>
          <form onSubmit={handleApply} style={{ display: 'flex', flexDirection: 'column', gap: 12 }}>
            {applyError && (
              <div role="alert" style={{ background: 'rgba(248,113,113,0.1)', border: '1px solid rgba(248,113,113,0.3)', borderRadius: 8, padding: 12, color: '#F87171', fontSize: 13 }}>
                {applyError}
              </div>
            )}
            <textarea
              value={applyMessage}
              onChange={e => setApplyMessage(e.target.value)}
              placeholder="Why are you qualified? (optional)"
              style={{
                width: '100%',
                padding: '10px 14px',
                borderRadius: 8,
                border: '1px solid var(--border-default)',
                backgroundColor: 'var(--bg-inset)',
                color: 'var(--text-primary)',
                fontSize: 14,
                minHeight: 80,
                resize: 'vertical',
                outline: 'none',
              }}
            />
            <button
              type="submit"
              disabled={applying}
              style={{
                padding: '10px 24px',
                borderRadius: 10,
                border: 'none',
                background: applying ? 'var(--text-tertiary)' : 'var(--accent-gradient)',
                color: '#000',
                fontWeight: 600,
                fontSize: 14,
                cursor: applying ? 'not-allowed' : 'pointer',
                alignSelf: 'flex-start',
              }}
            >
              {applying ? 'Applying...' : 'Sign & Apply'}
            </button>
          </form>
        </div>
      )}

      {/* Already applied notice */}
      {hasApplied && !isPoster && (
        <div className="card" style={{ background: 'rgba(52,211,153,0.04)' }}>
          <p style={{ color: 'var(--accent-primary)', fontSize: 14, display: 'flex', alignItems: 'center', gap: 8 }}>
            <CheckCircle size={16} /> You have applied to this bounty
          </p>
        </div>
      )}
    </div>
  );
}
```
